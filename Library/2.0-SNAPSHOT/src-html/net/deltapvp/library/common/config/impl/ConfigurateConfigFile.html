096    public ConfigurateConfigFile(final File configFile, final String templateName, final String header) {
097        this.configFile = configFile;
098        this.loader = YamlConfigurationLoader.builder()
099                .defaultOptions(opts -> opts
100                        .header(header)
101                        .serializers(SERIALIZERS))
102                .headerMode(HeaderMode.PRESET)
103                .nodeStyle(NodeStyle.BLOCK)
104                .indent(2)
105                .file(configFile)
106                .build();
107        this.templateName = templateName;
108    }

323    public CommentedConfigurationNode toSplitRoot(String path, final CommentedConfigurationNode node) {
324        if (path == null) {
325            return node;
326        }
327        path = path.startsWith(".") ? path.substring(1) : path;
328        return node.node(path.contains(".") ? path.split("\\.") : new Object[]{path});
329    }

331    public synchronized void load() {
332        if (pendingWrites.get() != 0) {
333            LibraryShared.getPlugin().getPluginLogger().info("Parsing config file {0} has been aborted due to {1} current pending write(s).", configFile, pendingWrites.get());
334            return;
335        }
336
337        if (configFile.getParentFile() != null && !configFile.getParentFile().exists()) {
338            if (!configFile.getParentFile().mkdirs()) {
339                LibraryShared.getPlugin().getPluginLogger().error("Failed to create config {0}.", configFile.toString());
340                return;
341            }
342        }
343
344        if (!configFile.exists()) {
345            if (templateName != null) {
346                try (final InputStream is = resourceClass.getResourceAsStream(templateName)) {
347                    LibraryShared.getPlugin().getPluginLogger().info("Creating config from template: {0}", configFile.toString());
348                    Files.copy(is, configFile.toPath());
349                } catch (IOException e) {
350                    LibraryShared.getPlugin().getPluginLogger().error("Failed to write config {0}.", e, configFile.toString());
351                }
352            }
353        }
354
355        try {
356            configurationNode = loader.load();
357        } catch (final ParsingException e) {
358            final File broken = new File(configFile.getAbsolutePath() + ".broken." + System.currentTimeMillis());
359            if (configFile.renameTo(broken)) {
360                LibraryShared.getPlugin().getPluginLogger().error("The file " + configFile.toString() + " is broken, it has been renamed to " + broken.toString(), e.getCause());
361                return;
362            }
363            LibraryShared.getPlugin().getPluginLogger().error("The file " + configFile.toString() + " is broken. A backup file has failed to be created", e.getCause());
364        } catch (final ConfigurateException e) {
365            LibraryShared.getPlugin().getPluginLogger().error(e.getMessage(), e);
366        } finally {
367            // Something is wrong! We need a node! I hope the backup worked!
368            if (configurationNode == null) {
369                configurationNode = loader.createNode();
370            }
371        }
372    }

374    /**
375     * Begins a transaction.
376     * <p>
377     * A transaction informs Essentials to pause the saving of data. This is should be used when
378     * bulk operations are being done and data shouldn't be saved until after the transaction has
379     * been completed.
380     */
381    public void startTransaction() {
382        transaction.set(true);
383    }
384
385    public void stopTransaction() {
386        stopTransaction(false);
387    }
388
389    public void stopTransaction(final boolean blocking) {
390        transaction.set(false);
391        if (blocking) {
392            blockingSave();
393        } else {
394            save();
395        }
396    }

398    public void setSaveHook(Runnable saveHook) {
399        this.saveHook = saveHook;
400    }
401
402    public synchronized void save() {
403        if (!transaction.get()) {
404            delaySave();
405        }
406    }
407
408    public synchronized void blockingSave() {
409        try {
410            delaySave().get();
411        } catch (final InterruptedException | ExecutionException e) {
412            LibraryShared.getPlugin().getPluginLogger().error(e.getMessage(), e);
413        }
414    }
415
416    private Future<?> delaySave() {
417        if (saveHook != null) {
418            saveHook.run();
419        }
420
421        final CommentedConfigurationNode node = configurationNode.copy();
422
423        pendingWrites.incrementAndGet();
424
425        return EXECUTOR_SERVICE.submit(new ConfigurationSaveTask(loader, node, pendingWrites));
426    }
