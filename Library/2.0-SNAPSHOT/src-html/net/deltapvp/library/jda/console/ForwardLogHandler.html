001package net.deltapvp.library.jda.console;
002
003import java.util.Map;
004import java.util.concurrent.ConcurrentHashMap;
005import java.util.logging.ConsoleHandler;
006import java.util.logging.Level;
007import java.util.logging.LogRecord;
008import org.apache.logging.log4j.LogManager;
009import org.apache.logging.log4j.Logger;
010
011public class ForwardLogHandler extends ConsoleHandler {
012    private Map<String, Logger> cachedLoggers = new ConcurrentHashMap<String, Logger>();
013
014    private Logger getLogger(String name) {
015        Logger logger = this.cachedLoggers.get(name);
016        if (logger == null) {
017            logger = LogManager.getLogger(name);
018            this.cachedLoggers.put(name, logger);
019        }
020
021        return logger;
022    }
023
024    @Override
025    public void publish(LogRecord record) {
026        Logger logger = this.getLogger(String.valueOf(record.getLoggerName())); // See SPIGOT-1230
027        Throwable exception = record.getThrown();
028        Level level = record.getLevel();
029        String message = getFormatter().formatMessage(record);
030
031        if (level == Level.SEVERE) {
032            logger.error(message, exception);
033        } else if (level == Level.WARNING) {
034            logger.warn(message, exception);
035        } else if (level == Level.INFO) {
036            logger.info(message, exception);
037        } else if (level == Level.CONFIG) {
038            logger.debug(message, exception);
039        } else {
040            logger.trace(message, exception);
041        }
042    }
043
044    @Override
045    public void flush() {}
046
047    @Override
048    public void close() throws SecurityException {}
049}
