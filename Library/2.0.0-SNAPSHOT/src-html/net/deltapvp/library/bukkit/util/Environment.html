025package net.deltapvp.library.bukkit.util;
026
027import org.bukkit.Bukkit;
028
029public class Environment {
030    private static final boolean paper;
031    private static final String serverPackageName;
032    private static final String serverApiVersion;
033    private static final int majorMinecraftVersion;
034
035    static {
036        paper = classExists("com.destroystokyo.paper.PaperConfig");
037        serverPackageName = Bukkit.getServer().getClass().getPackage().getName();
038        serverApiVersion = serverPackageName.substring(serverPackageName.lastIndexOf('.') + 1);
039        majorMinecraftVersion = Integer.parseInt(serverApiVersion.split("_")[1]);
040    }
041
042    public static boolean classExists(final String fullyQualifiedName) {
043        try {
044            Class.forName(fullyQualifiedName);
045            return true;
046        } catch (ClassNotFoundException e) {
047            return false;
048        }
049    }
050
051    public static boolean paper() {
052        return Environment.paper;
053    }
054
055    public static String serverPackageName() {
056        return Environment.serverPackageName;
057    }
058
059    public static String serverApiVersion() {
060        return Environment.serverApiVersion;
061    }
062
063    public static int majorMinecraftVersion() {
064        return Environment.majorMinecraftVersion;
065    }
066}
