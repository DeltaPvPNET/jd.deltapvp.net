061 * Contains all the common methods for the future ItemBuilders

068    private static final EnumSet<Material> LEATHER_ARMOR = EnumSet.of(
069            Material.LEATHER_HELMET, Material.LEATHER_CHESTPLATE, Material.LEATHER_LEGGINGS, Material.LEATHER_BOOTS
070    );

072    private static final GsonComponentSerializer GSON = Instances.GSON_SERIALIZER;
073    private static final Field DISPLAY_NAME_FIELD;
074    private static final Field LORE_FIELD;
075
076    static {
077        try {
078            final Class<?> metaClass = VersionHelper.craftClass("inventory.CraftMetaItem");
079
080            DISPLAY_NAME_FIELD = metaClass.getDeclaredField("displayName");
081            DISPLAY_NAME_FIELD.setAccessible(true);
082
083            LORE_FIELD = metaClass.getDeclaredField("lore");
084            LORE_FIELD.setAccessible(true);
085        } catch (NoSuchFieldException | ClassNotFoundException exception) {
086            exception.printStackTrace();
087            throw new RuntimeException("Could not retrieve displayName nor lore field for ItemBuilder.");
088        }
089    }

093
094    protected BaseItemBuilder(@NotNull final ItemStack itemStack) {
095        Validate.notNull(itemStack, "Item can't be null!");
096
097        this.itemStack = itemStack;
098        meta = itemStack.hasItemMeta() ? itemStack.getItemMeta() : Bukkit.getItemFactory().getItemMeta(itemStack.getType());
099    }

101    /**
102     * Sets the display name of the item using {@link Component}
103     *
104     * @param name The {@link Component} name
105     * @return {@link ItemBuilder}
106     * @since 3.0.0
107     */
108    @NotNull
109    @Contract("_ -> this")
110    public B name(@NotNull final Component name) {
111        if (meta == null) return (B) this;
112
113        if (VersionHelper.IS_ITEM_LEGACY) {
114            meta.setDisplayName(TextUtil.unparse(name, MessageType.LEGACY_STUPID));
115            return (B) this;
116        }
117
118        try {
119            DISPLAY_NAME_FIELD.set(meta, GSON.serialize(name));
120        } catch (IllegalAccessException exception) {
121            exception.printStackTrace();
122        }
123
124        return (B) this;
125    }

154    /**
155     * Set the lore lines of an item
156     *
157     * @param lore A {@link List} with the lore lines
158     * @return {@link ItemBuilder}
159     * @since 3.0.0
160     */
161    @NotNull
162    @Contract("_ -> this")
163    public B lore(@NotNull final List<@Nullable Component> lore) {
164        if (meta == null) return (B) this;
165
166        if (VersionHelper.IS_ITEM_LEGACY) {
167            meta.setLore(lore.stream().filter(Objects::nonNull).map(Instances.LEGACY_SERIALIZER_STUPID::serialize).collect(Collectors.toList()));
168            return (B) this;
169        }
170
171        final List<String> jsonLore = lore.stream().filter(Objects::nonNull).map(GSON::serialize).collect(Collectors.toList());
172
173        try {
174            LORE_FIELD.set(meta, jsonLore);
175        } catch (IllegalAccessException exception) {
176            exception.printStackTrace();
177        }
178
179        return (B) this;
180    }
181
182    /**
183     * Consumer for freely adding to the lore
184     *
185     * @param lore A {@link Consumer} with the {@link List} of lore {@link Component}
186     * @return {@link ItemBuilder}
187     * @since 3.0.0
188     */
189    @NotNull
190    @Contract("_ -> this")
191    public B lore(@NotNull final Consumer<List<@Nullable Component>> lore) {
192        if (meta == null) return (B) this;
193
194        List<Component> components;
195        if (VersionHelper.IS_ITEM_LEGACY) {
196            final List<String> stringLore = meta.getLore();
197            if (stringLore == null) return (B) this;
198            components = stringLore.stream().map(Instances.LEGACY_SERIALIZER_STUPID::deserialize).collect(Collectors.toList());
199        } else {
200            try {
201                final List<String> jsonLore = (List<String>) LORE_FIELD.get(meta);
202                components = jsonLore.stream().map(GSON::deserialize).collect(Collectors.toList());
203            } catch (IllegalAccessException exception) {
204                components = EmptyUtils.EMPTY_COMP_LIST;
205                exception.printStackTrace();
206            }
207        }
208
209        lore.accept(components);
210        return lore(components);
211    }

325    /**
326     * Adds or removes the {@link ItemStack} glow
327     *
328     * @param glow Should the item glow
329     * @return {@link ItemBuilder}
330     */
331    @NotNull
332    @Contract("_ -> this")
333    public B glow(boolean glow) {
334        if (glow) {
335            meta.addEnchant(Enchantment.LURE, 1, false);
336            meta.addItemFlags(ItemFlag.HIDE_ENCHANTS);
337            return (B) this;
338        }
339
340        for (final Enchantment enchantment : meta.getEnchants().keySet()) {
341            meta.removeEnchant(enchantment);
342        }
343
344        return (B) this;
345    }

394    /**
395     * Color an {@link org.bukkit.inventory.ItemStack}
396     *
397     * @param color color
398     * @return {@link B}
399     * @see org.bukkit.inventory.meta.LeatherArmorMeta#setColor(Color)
400     * @see org.bukkit.inventory.meta.MapMeta#setColor(Color)
401     * @since 3.0.3
402     */
403    @NotNull
404    @Contract("_ -> this")
405    public B color(@NotNull final Color color) {
406        if (LEATHER_ARMOR.contains(itemStack.getType())) {
407            final LeatherArmorMeta leatherArmorMeta = (LeatherArmorMeta) getMeta();
408
409            leatherArmorMeta.setColor(color);
410            setMeta(leatherArmorMeta);
411        }
412
413        return (B) this;
414    }

416    /**
417     * Sets NBT tag to the {@link ItemStack}
418     *
419     * @param key   The NBT key
420     * @param value The NBT value
421     * @return {@link ItemBuilder}
422     */
423    @NotNull
424    @Contract("_, _ -> this")
425    public B setNbt(@NotNull final String key, @Nullable final String value) {
426        itemStack.setItemMeta(meta);
427        itemStack = ItemNbt.setString(itemStack, key, value);
428        meta = itemStack.getItemMeta();
429        return (B) this;
430    }

439    @NotNull
440    @Contract("_, _ -> this")
441    public B setNbt(@NotNull final String key, final boolean value) {
442        itemStack.setItemMeta(meta);
443        itemStack = ItemNbt.setBoolean(itemStack, key, value);
444        meta = itemStack.getItemMeta();
445        return (B) this;
