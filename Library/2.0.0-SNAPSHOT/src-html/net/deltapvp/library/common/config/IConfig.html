025package net.deltapvp.library.common.config;
026
027import java.util.List;
028
029import com.google.common.collect.Lists;
030
031import net.kyori.adventure.text.Component;
032public interface IConfig<T, F> {
033
034    String getName();
035
036    T getHandle();
037
038    String getString(String path, String def);
039
040    default String getString(String path) {
041        return getString(path, null);
042    }
043
044    int getInt(String path, int def);
045
046    default int getInt(String path) {
047        return getInt(path, 0);
048    }
049
050    long getLong(String path, long def);
051
052    default long getLong(String path) {
053        return getLong(path, 0L);
054    }
055
056    double getDouble(String path, double def);
057
058    default double getDouble(String path) {
059        return getDouble(path, 0.0D);
060    }
061
062    float getFloat(String path, float def);
063
064    default float getFloat(String path) {
065        return getFloat(path, 0.0F);
066    }
067
068    boolean getBoolean(String path, boolean def);
069
070    default boolean getBoolean(String path) {
071        return getBoolean(path, false);
072    }
073
074    Object get(String path, Object def);
075
076    default Object get(String path) {
077        return get(path, null);
078    }
079
080    List<String> getStringList(String path, List<String> def);
081
082    default List<String> getStringList(String path) {
083        return getStringList(path, Lists.newArrayList());
084    }
085
086    List<?> getList(String path, List<?> def);
087
088    default List<?> getList(String path) {
089        return getList(path, Lists.newArrayList());
090    }
091
092    List<Component> getComponentList(String path, List<Component> def);
093    default List<Component> getComponentList(String path) {
094        return getComponentList(path, Lists.newArrayList());
095    }
096
097    Component getComponent(String path, Component def);
098
099    default Component getComponent(String path) {
100        return getComponent(path, Component.empty());
101    }
102
103    void set(String path, Object object);
104    
105    F getSection(String path);
106
107    void reload();
108}
