025package net.deltapvp.library.common.loader;
026
027import java.net.URL;
028import java.net.URLClassLoader;
029
030/**
031 * A classloader "isolated" from the rest of the Minecraft server.
032 *
033 * <p>Used to load specific LuckPerms dependencies without causing conflicts
034 * with other plugins, or libraries provided by the server implementation.</p>
035 */
036public class IsolatedClassLoader extends URLClassLoader {
037    static {
038        ClassLoader.registerAsParallelCapable();
039    }
040
041    public IsolatedClassLoader(URL[] urls) {
042        /*
043         * ClassLoader#getSystemClassLoader returns the AppClassLoader
044         *
045         * Calling #getParent on this returns the ExtClassLoader (Java 8) or
046         * the PlatformClassLoader (Java 9). Since we want this classloader to
047         * be isolated from the Minecraft server (the app), we set the parent
048         * to be the platform class loader.
049         */
050        super(urls, ClassLoader.getSystemClassLoader().getParent());
051    }
052
053}
