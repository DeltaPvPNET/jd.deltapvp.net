025package net.deltapvp.library.common.logging;
026
027import net.deltapvp.library.common.util.EmptyUtils;
028
029/**
030 * Represents the logger instance being used by Library on the platform.
031 *
032 * <p>
033 * Messages sent using the logger are sent prefixed with the Library tag, and on some
034 * implementations will be colored depending on the message type.
035 * </p>
036 */
037public interface PluginLogger {
038
039    default void info(String s) {
040        info(s, null, EmptyUtils.EMPTY_OBJ_VARARG);
041    }
042
043    default void info(String s, Throwable t) {
044        info(s, t, EmptyUtils.EMPTY_OBJ_VARARG);
045    }
046
047    default void info(String s, Object... arguments) {
048        info(s, null, arguments);
049    }
050
051    void info(String s, Throwable t, Object... arguments);
052
053    default void warn(String s) {
054        warn(s, null, EmptyUtils.EMPTY_OBJ_VARARG);
055    }
056
057    default void warn(String s, Throwable t) {
058        warn(s, t, EmptyUtils.EMPTY_OBJ_VARARG);
059    }
060
061    default void warn(String s, Object... arguments) {
062        warn(s, null, arguments);
063    }
064
065    void warn(String s, Throwable t, Object... arguments);
066
067    default void error(String s) {
068        error(s, null, EmptyUtils.EMPTY_OBJ_VARARG);
069    }
070
071    default void error(String s, Throwable t) {
072        error(s, t, EmptyUtils.EMPTY_OBJ_VARARG);
073    }
074
075    default void error(String s, Object... arguments) {
076        error(s, null, arguments);
077    }
078
079    void error(String s, Throwable t, Object... arguments);
080
081    default void debug(String s) {
082        debug(s, null, EmptyUtils.EMPTY_OBJ_VARARG);
083    }
084
085    default void debug(String s, Throwable t) {
086        debug(s, t, EmptyUtils.EMPTY_OBJ_VARARG);
087    }
088
089    default void debug(String s, Object... arguments) {
090        debug(s, null, arguments);
091    }
092
093    void debug(String s, Throwable t, Object... arguments);
094
095    default String getMessage(String string, Object... arguments) {
096        String finalMessage = string;
097        for (int i = 0; i < arguments.length; i++) {
098            finalMessage = finalMessage.replace("{" + i + "}", arguments[i].toString());
099        }
100
101        return finalMessage;
102    }
103}
