025package net.deltapvp.library.common.util.adapters;
026
027import java.io.IOException;
028
029import com.google.gson.TypeAdapter;
030import com.google.gson.stream.JsonReader;
031import com.google.gson.stream.JsonWriter;
032
033import org.jetbrains.annotations.Nullable;
034
035import net.deltapvp.library.common.util.text.MessageType;
036import net.deltapvp.library.common.util.text.TextUtil;
037import net.kyori.adventure.text.Component;
038
039public class ComponentSerializer extends TypeAdapter<Component> {
040
041    @Override
042    public void write(final JsonWriter jsonWriter, final @Nullable Component component) throws IOException {
043        if (component != null) {
044            jsonWriter.value(TextUtil.unparse(component, MessageType.GSON));
045            //jsonWriter.value(Instances.GSON_SERIALIZER.serialize(component));
046        } else {
047            jsonWriter.value((String) null);
048        }
049    }
050
051    @Override
052    public Component read(final JsonReader jsonReader) throws IOException {
053        return TextUtil.parse(jsonReader.nextString(), MessageType.GSON);
054        //return Instances.GSON_SERIALIZER.deserialize(jsonReader.nextString());
055    }
056
057}
