025package net.deltapvp.library.common.util.json;
026
027import java.util.List;
028import java.util.Map;
029import java.util.Set;
030import java.util.UUID;
031import java.util.Optional;
032
033import com.google.gson.JsonElement;
034import com.google.gson.JsonObject;
035
036import net.deltapvp.library.common.util.FastUUID;
037import net.deltapvp.library.common.util.Instances;
038
039public class JsonChain {
040
041    private final JsonObject json = new JsonObject();
042
043    public JsonChain addProperty(String property, String value) {
044        this.json.addProperty(property, value);
045        return this;
046    }
047
048    public JsonChain addProperty(String property, Number value) {
049        this.json.addProperty(property, value);
050        return this;
051    }
052
053    public JsonChain addProperty(String property, Boolean value) {
054        this.json.addProperty(property, value);
055        return this;
056    }
057
058    public JsonChain addProperty(String property, Character value) {
059        this.json.addProperty(property, value);
060        return this;
061    }
062
063    public JsonChain addProperty(String property, Object value) {
064        this.json.addProperty(property, Instances.GSON.toJson(value));
065        return this;
066    }
067
068    public JsonChain addProperty(String property, UUID value) {
069        this.json.addProperty(property, FastUUID.toString(value));
070        return this;
071    }
072
073    public <U> JsonChain addProperty(String property, List<U> value) {
074        this.json.addProperty(property, Instances.GSON.toJson(value));
075        return this;
076    }
077
078    public <U> JsonChain addProperty(String property, Set<U> value) {
079        this.json.addProperty(property, Instances.GSON.toJson(value));
080        return this;
081    }
082
083    public <K, V> JsonChain addProperty(String property, Map<K, V> value) {
084        this.json.addProperty(property, Instances.GSON.toJson(value));
085        return this;
086    }
087
088    public <U> JsonChain addProperty(String property, Optional<U> value) {
089        return addProperty(property, value, true);
090    }
091
092    public <U> JsonChain addProperty(String property, Optional<U> value, boolean serializeNull) {
093        if (serializeNull) {
094            this.json.addProperty(property, value.isPresent() ? Instances.GSON.toJson(value.get()) : null);
095        } else {
096            value.ifPresent(v -> this.json.addProperty(property, Instances.GSON.toJson(v)));
097        }
098        return this;
099    }
100
101    public JsonChain add(String property, JsonElement element) {
102        this.json.add(property, element);
103        return this;
104    }
105
106    public JsonChain add(String property, JsonObject object) {
107        this.json.add(property, object);
108        return this;
109    }
110
111    public JsonObject get() {
112        return this.json;
113    }
114
115    public String toJson() {
116        return Instances.GSON.toJson(get());
117    }
118
119}
