025package net.deltapvp.library.common.util.tuples;
026
027import org.jetbrains.annotations.NotNull;
028
029import java.util.Objects;
030
031/**
032 * Immutable generic 2-tuple
033 *
034 * @param <U> First type
035 * @param <V> Second type
036 */
037public class Pair<U, V> implements Tuple {
038
039    private final U first;
040    private final V second;
041
042    protected Pair(
043            final U first,
044            final V second
045    ) {
046        this.first = first;
047        this.second = second;
048    }
049
050    /**
051     * Create a new 2-tuple
052     *
053     * @param first  First value
054     * @param second Second value
055     * @param <U>    First type
056     * @param <V>    Second type
057     * @return Created pair
058     */
059    public static <U, V> @NotNull Pair<U, V> of(
060            final U first,
061            final V second
062    ) {
063        return new Pair<>(first, second);
064    }
065
066    /**
067     * Get the first value
068     *
069     * @return First value
070     */
071    public final U getFirst() {
072        return this.first;
073    }
074
075    /**
076     * Get the second value
077     *
078     * @return Second value
079     */
080    public final V getSecond() {
081        return this.second;
082    }
083
084    @Override
085    public final boolean equals(final Object o) {
086        if (this == o) {
087            return true;
088        }
089        if (o == null || getClass() != o.getClass()) {
090            return false;
091        }
092        final Pair<?, ?> pair = (Pair<?, ?>) o;
093        return Objects.equals(this.getFirst(), pair.getFirst())
094                && Objects.equals(this.getSecond(), pair.getSecond());
095    }
096
097    @Override
098    public final int hashCode() {
099        return Objects.hash(this.getFirst(), this.getSecond());
100    }
101
102    @Override
103    public final String toString() {
104        return String.format("(%s, %s)", this.first, this.second);
105    }
106
107    @Override
108    public final int getSize() {
109        return 2;
110    }
111
112    @Override
113    public final @NotNull Object @NotNull [] toArray() {
114        final Object[] array = new Object[2];
115        array[0] = this.first;
116        array[1] = this.second;
117        return array;
118    }
119
120}
