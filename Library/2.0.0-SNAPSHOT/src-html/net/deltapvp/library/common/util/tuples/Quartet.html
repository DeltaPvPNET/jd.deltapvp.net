025package net.deltapvp.library.common.util.tuples;
026
027import org.jetbrains.annotations.NotNull;
028
029import java.util.Objects;
030
031/**
032 * Immutable generic 5-tuple
033 *
034 * @param <U> First type
035 * @param <V> Second type
036 * @param <W> Third type
037 * @param <X> Fourth type
038 */
039public class Quartet<U, V, W, X> implements Tuple {
040
041    private final U first;
042    private final V second;
043    private final W third;
044    private final X fourth;
045
046    protected Quartet(
047            final @NotNull U first,
048            final @NotNull V second,
049            final @NotNull W third,
050            final @NotNull X fourth
051    ) {
052        this.first = first;
053        this.second = second;
054        this.third = third;
055        this.fourth = fourth;
056    }
057
058    /**
059     * Create a new 4-tuple
060     *
061     * @param first  First value
062     * @param second Second value
063     * @param third  Third value
064     * @param fourth Fourth value
065     * @param <U>    First type
066     * @param <V>    Second type
067     * @param <W>    Third type
068     * @param <X>    Fourth type
069     * @return Created quartet
070     */
071    public static <U, V, W, X> @NotNull Quartet<@NotNull U, @NotNull V, @NotNull W, @NotNull X> of(
072            final @NotNull U first,
073            final @NotNull V second,
074            final @NotNull W third,
075            final @NotNull X fourth
076    ) {
077        return new Quartet<>(first, second, third, fourth);
078    }
079
080    /**
081     * Get the first value
082     *
083     * @return First value
084     */
085    public final @NotNull U getFirst() {
086        return this.first;
087    }
088
089    /**
090     * Get the second value
091     *
092     * @return Second value
093     */
094    public final @NotNull V getSecond() {
095        return this.second;
096    }
097
098    /**
099     * Get the third value
100     *
101     * @return Third value
102     */
103    public final @NotNull W getThird() {
104        return this.third;
105    }
106
107    /**
108     * Get the fourth value
109     *
110     * @return Fourth value
111     */
112    public final @NotNull X getFourth() {
113        return this.fourth;
114    }
115
116    @Override
117    public final boolean equals(final Object o) {
118        if (this == o) {
119            return true;
120        }
121        if (o == null || getClass() != o.getClass()) {
122            return false;
123        }
124        final Quartet<?, ?, ?, ?> quartet = (Quartet<?, ?, ?, ?>) o;
125        return Objects.equals(this.getFirst(), quartet.getFirst())
126                && Objects.equals(this.getSecond(), quartet.getSecond())
127                && Objects.equals(this.getThird(), quartet.getThird())
128                && Objects.equals(this.getFourth(), quartet.getFourth());
129    }
130
131    @Override
132    public final int hashCode() {
133        return Objects.hash(this.getFirst(), this.getSecond(), this.getThird(), this.getFourth());
134    }
135
136    @Override
137    public final String toString() {
138        return String.format("(%s, %s, %s, %s)", this.first, this.second, this.third, this.fourth);
139    }
140
141    @Override
142    public final int getSize() {
143        return 4;
144    }
145
146    @Override
147    public final @NotNull Object @NotNull [] toArray() {
148        final Object[] array = new Object[4];
149        array[0] = this.first;
150        array[1] = this.second;
151        array[2] = this.third;
152        array[3] = this.fourth;
153        return array;
154    }
155
156}
