025package net.deltapvp.library.common.util.tuples;
026
027import org.jetbrains.annotations.NotNull;
028
029import java.util.Objects;
030
031/**
032 * Immutable generic 5-tuple
033 *
034 * @param <U> First type
035 * @param <V> Second type
036 * @param <W> Third type
037 * @param <X> Fourth type
038 * @param <Y> Fifth type
039 */
040public class Quintet<U, V, W, X, Y> implements Tuple {
041
042    private final U first;
043    private final V second;
044    private final W third;
045    private final X fourth;
046    private final Y fifth;
047
048    protected Quintet(
049            final @NotNull U first,
050            final @NotNull V second,
051            final @NotNull W third,
052            final @NotNull X fourth,
053            final @NotNull Y fifth
054    ) {
055        this.first = first;
056        this.second = second;
057        this.third = third;
058        this.fourth = fourth;
059        this.fifth = fifth;
060    }
061
062    /**
063     * Create a new 5-tuple
064     *
065     * @param first  First value
066     * @param second Second value
067     * @param third  Third value
068     * @param fourth Fourth value
069     * @param fifth  fifth value
070     * @param <U>    First type
071     * @param <V>    Second type
072     * @param <W>    Third type
073     * @param <X>    Fourth type
074     * @param <Y>    Fifth type
075     * @return Created quintet
076     */
077    public static <U, V, W, X, Y> @NotNull Quintet<@NotNull U, @NotNull V, @NotNull W, @NotNull X, @NotNull Y> of(
078            final @NotNull U first,
079            final @NotNull V second,
080            final @NotNull W third,
081            final @NotNull X fourth,
082            final @NotNull Y fifth
083    ) {
084        return new Quintet<>(first, second, third, fourth, fifth);
085    }
086
087    /**
088     * Get the first value
089     *
090     * @return First value
091     */
092    public final @NotNull U getFirst() {
093        return this.first;
094    }
095
096    /**
097     * Get the second value
098     *
099     * @return Second value
100     */
101    public final @NotNull V getSecond() {
102        return this.second;
103    }
104
105    /**
106     * Get the third value
107     *
108     * @return Third value
109     */
110    public final @NotNull W getThird() {
111        return this.third;
112    }
113
114    /**
115     * Get the fourth value
116     *
117     * @return Fourth value
118     */
119    public final @NotNull X getFourth() {
120        return this.fourth;
121    }
122
123    /**
124     * Get the fifth value
125     *
126     * @return Fifth value
127     */
128    public final @NotNull Y getFifth() {
129        return this.fifth;
130    }
131
132    @Override
133    public final boolean equals(final Object o) {
134        if (this == o) {
135            return true;
136        }
137        if (o == null || getClass() != o.getClass()) {
138            return false;
139        }
140        final Quintet<?, ?, ?, ?, ?> quintet = (Quintet<?, ?, ?, ?, ?>) o;
141        return Objects.equals(this.getFirst(), quintet.getFirst())
142                && Objects.equals(this.getSecond(), quintet.getSecond())
143                && Objects.equals(this.getThird(), quintet.getThird())
144                && Objects.equals(this.getFourth(), quintet.getFourth())
145                && Objects.equals(this.getFifth(), quintet.getFifth());
146    }
147
148    @Override
149    public final int hashCode() {
150        return Objects.hash(this.getFirst(), this.getSecond(), this.getThird(), this.getFourth(), this.getFifth());
151    }
152
153    @Override
154    public final String toString() {
155        return String.format("(%s, %s, %s, %s, %s)", this.first, this.second, this.third, this.fourth, this.fifth);
156    }
157
158    @Override
159    public final int getSize() {
160        return 5;
161    }
162
163    @Override
164    public final @NotNull Object @NotNull [] toArray() {
165        final Object[] array = new Object[5];
166        array[0] = this.first;
167        array[1] = this.second;
168        array[2] = this.third;
169        array[3] = this.fourth;
170        array[4] = this.fifth;
171        return array;
172    }
173
174}
