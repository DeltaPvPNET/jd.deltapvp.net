025package net.deltapvp.library.common.util.tuples;
026
027import org.jetbrains.annotations.NotNull;
028
029import java.util.Objects;
030
031/**
032 * Immutable generic 6-tuple
033 *
034 * @param <U> First type
035 * @param <V> Second type
036 * @param <W> Third type
037 * @param <X> Fourth type
038 * @param <Y> Fifth type
039 * @param <Z> Sixth type
040 */
041public class Sextet<U, V, W, X, Y, Z> implements Tuple {
042
043    private final U first;
044    private final V second;
045    private final W third;
046    private final X fourth;
047    private final Y fifth;
048    private final Z sixth;
049
050    protected Sextet(
051            final @NotNull U first,
052            final @NotNull V second,
053            final @NotNull W third,
054            final @NotNull X fourth,
055            final @NotNull Y fifth,
056            final @NotNull Z sixth
057    ) {
058        this.first = first;
059        this.second = second;
060        this.third = third;
061        this.fourth = fourth;
062        this.fifth = fifth;
063        this.sixth = sixth;
064    }
065
066    /**
067     * Create a new 6-tuple
068     *
069     * @param first  First value
070     * @param second Second value
071     * @param third  Third value
072     * @param fourth Fourth value
073     * @param fifth  fifth value
074     * @param sixth  Sixth value
075     * @param <U>    First type
076     * @param <V>    Second type
077     * @param <W>    Third type
078     * @param <X>    Fourth type
079     * @param <Y>    Fifth type
080     * @param <Z>    Sixth type
081     * @return Created sextet
082     */
083    public static <U, V, W, X, Y, Z> @NotNull Sextet<@NotNull U, @NotNull V, @NotNull W, @NotNull X, @NotNull Y, @NotNull Z> of(
084            final @NotNull U first,
085            final @NotNull V second,
086            final @NotNull W third,
087            final @NotNull X fourth,
088            final @NotNull Y fifth,
089            final @NotNull Z sixth
090    ) {
091        return new Sextet<>(first, second, third, fourth, fifth, sixth);
092    }
093
094    /**
095     * Get the first value
096     *
097     * @return First value
098     */
099    public final @NotNull U getFirst() {
100        return this.first;
101    }
102
103    /**
104     * Get the second value
105     *
106     * @return Second value
107     */
108    public final @NotNull V getSecond() {
109        return this.second;
110    }
111
112    /**
113     * Get the third value
114     *
115     * @return Third value
116     */
117    public final @NotNull W getThird() {
118        return this.third;
119    }
120
121    /**
122     * Get the fourth value
123     *
124     * @return Fourth value
125     */
126    public final @NotNull X getFourth() {
127        return this.fourth;
128    }
129
130    /**
131     * Get the fifth value
132     *
133     * @return Fifth value
134     */
135    public final @NotNull Y getFifth() {
136        return this.fifth;
137    }
138
139    /**
140     * Get the sixth value
141     *
142     * @return Sixth value
143     */
144    public final @NotNull Z getSixth() {
145        return this.sixth;
146    }
147
148    @Override
149    public final boolean equals(final Object o) {
150        if (this == o) {
151            return true;
152        }
153        if (o == null || getClass() != o.getClass()) {
154            return false;
155        }
156        final Sextet<?, ?, ?, ?, ?, ?> sextet = (Sextet<?, ?, ?, ?, ?, ?>) o;
157        return Objects.equals(this.getFirst(), sextet.getFirst())
158                && Objects.equals(this.getSecond(), sextet.getSecond())
159                && Objects.equals(this.getThird(), sextet.getThird())
160                && Objects.equals(this.getFourth(), sextet.getFourth())
161                && Objects.equals(this.getFifth(), sextet.getFifth())
162                && Objects.equals(this.getSixth(), sextet.getSixth());
163    }
164
165    @Override
166    public final int hashCode() {
167        return Objects.hash(this.getFirst(), this.getSecond(), this.getThird(), this.getFourth(), this.getFifth(), this.getSixth());
168    }
169
170    @Override
171    public final String toString() {
172        return String.format("(%s, %s, %s, %s, %s, %s)", this.first, this.second, this.third,
173                this.fourth, this.fifth, this.sixth
174        );
175    }
176
177    @Override
178    public final int getSize() {
179        return 6;
180    }
181
182    @Override
183    public final @NotNull Object @NotNull [] toArray() {
184        final Object[] array = new Object[6];
185        array[0] = this.first;
186        array[1] = this.second;
187        array[2] = this.third;
188        array[3] = this.fourth;
189        array[4] = this.fifth;
190        array[5] = this.sixth;
191        return array;
192    }
193
194}
