025package net.deltapvp.library.common.util.tuples;
026
027import org.jetbrains.annotations.NotNull;
028
029import java.util.Objects;
030
031/**
032 * Immutable generic 3-tuple
033 *
034 * @param <U> First type
035 * @param <V> Second type
036 * @param <W> Third type
037 */
038public class Triplet<U, V, W> implements Tuple {
039
040    private final U first;
041    private final V second;
042    private final W third;
043
044    protected Triplet(
045            final @NotNull U first,
046            final @NotNull V second,
047            final @NotNull W third
048    ) {
049        this.first = first;
050        this.second = second;
051        this.third = third;
052    }
053
054    /**
055     * Create a new 3-tuple
056     *
057     * @param first  First value
058     * @param second Second value
059     * @param third  Third value
060     * @param <U>    First type
061     * @param <V>    Second type
062     * @param <W>    Third type
063     * @return Created triplet
064     */
065    public static <U, V, W> @NotNull Triplet<@NotNull U, @NotNull V, @NotNull W> of(
066            final @NotNull U first,
067            final @NotNull V second,
068            final @NotNull W third
069    ) {
070        return new Triplet<>(first, second, third);
071    }
072
073    /**
074     * Get the first value
075     *
076     * @return First value
077     */
078    public final U getFirst() {
079        return this.first;
080    }
081
082    /**
083     * Get the second value
084     *
085     * @return Second value
086     */
087    public final V getSecond() {
088        return this.second;
089    }
090
091    /**
092     * Get the third value
093     *
094     * @return Third value
095     */
096    public final W getThird() {
097        return this.third;
098    }
099
100    @Override
101    public final boolean equals(final Object o) {
102        if (this == o) {
103            return true;
104        }
105        if (o == null || getClass() != o.getClass()) {
106            return false;
107        }
108        final Triplet<?, ?, ?> triplet = (Triplet<?, ?, ?>) o;
109        return Objects.equals(this.getFirst(), triplet.getFirst())
110                && Objects.equals(this.getSecond(), triplet.getSecond())
111                && Objects.equals(this.getThird(), triplet.getThird());
112    }
113
114    @Override
115    public final int hashCode() {
116        return Objects.hash(this.getFirst(), this.getSecond(), this.getThird());
117    }
118
119    @Override
120    public final String toString() {
121        return String.format("(%s, %s, %s)", this.first, this.second, this.third);
122    }
123
124    @Override
125    public final int getSize() {
126        return 3;
127    }
128
129    @Override
130    public final @NotNull Object @NotNull [] toArray() {
131        final Object[] array = new Object[3];
132        array[0] = this.first;
133        array[1] = this.second;
134        array[2] = this.third;
135        return array;
136    }
137
138}
