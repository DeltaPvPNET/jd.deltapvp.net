001package net.deltapvp.library.jda;
002
003import java.text.DecimalFormat;
004import java.util.concurrent.CountDownLatch;
005
006public abstract class AbstractLauncher {
007    private static final CountDownLatch shutdownLatch = new CountDownLatch(1);
008    private static final CountDownLatch loadLatch = new CountDownLatch(1);
009    private static final CountDownLatch enableLatch = new CountDownLatch(1);
010
011    protected static <T extends DiscordBot> void main(String[] args, T bot) {
012        long startTime = System.currentTimeMillis();
013        System.setProperty("java.util.logging.manager", "org.apache.logging.log4j.jul.LogManager");
014        try {
015            Runtime.getRuntime().addShutdownHook(new Thread(() -> bot.onDisable(), "Shutdown thread"));
016        } finally {
017            shutdownLatch.countDown();
018        }
019        try {
020            bot.onLoad();
021        } finally {
022            loadLatch.countDown();
023        }
024        try {
025            bot.onEnable();
026        } finally {
027            enableLatch.countDown();
028        }
029
030        
031        double bootTime = (System.currentTimeMillis() - startTime) / 1000d;
032        bot.getPluginLogger().info("Done ({}s)!", new DecimalFormat("#.##").format(bootTime));
033    }
034    
035}
