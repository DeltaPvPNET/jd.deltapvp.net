040public abstract class DiscordBot implements LibraryPlugin {
041
042    protected final LibraryShared SHARED = new LibraryShared(this);
043    protected JDA jda;
044    protected PluginLogger logger;
045    private final EventWaiter eventWaiter = new EventWaiter();
046    private final AtomicBoolean shutdownInProgress = new AtomicBoolean(false);
047    private boolean shutdown = false;
048    private Logger log4jLogger = LogManager.getLogger("Discord");
049    protected SchedulerAdapter scheduler;
050    protected boolean isEnabled = false;
051    protected DiscordPlatformUtil platformUtil;
052    protected Thread mainThread;
053
054    public void onLoad() {
055        isEnabled = true;
056        setupMisc();
057        setupLogger();
058        setupConsole();
059        SHARED.onLoad();
060        setupJDA();
061        load();
062    }
063
064    public void onEnable() {
065        SHARED.onEnable();
066        enable();
067        this.jda.getPresence().setStatus(OnlineStatus.ONLINE);
068        this.jda.getPresence().setActivity(Activity.playing("with code"));
069    }
070
071    public void onDisable() {
072        if (!shutdownInProgress.compareAndSet(false, true)) {
073            return;
074        }
075        boolean explicitExit = true;
076
077        Runnable shutdownProcess = () -> {
078            this.jda.getPresence().setStatus(OnlineStatus.IDLE);
079            this.jda.getPresence().setActivity(Activity.watching("Shutdown"));
080            logger.info("Shutting down the bot...");
081            shutdown = true;
082            isEnabled = false;
083            disable();
084            scheduler.shutdownScheduler();
085            this.jda.shutdown();
086            SHARED.onDisable();
087            LogManager.shutdown();
088            scheduler.shutdownExecutor();
089            if (explicitExit) {
090                AccessController.doPrivileged(new PrivilegedAction<Void>() {
091                    @Override
092                    public Void run() {
093                        System.exit(0);
094                        return null;
095                    }
096                });
097            }
098        };
099        shutdownProcess.run();
100    }
101
102    @Override
103    public String getPluginName() {
104        return "Discord Bot";
105    }
106
107    private void setupMisc() {
108        try {
109            FileUtil.createDirectoriesIfNotExists(getDataDirectory().resolve("downloads"));
110        } catch (Throwable thr) {
111
112        }
113    }
114
115    @Override
116    public PluginLogger getPluginLogger() {
117        if (logger == null) {
118            logger = new Log4jPluginLogger(this.log4jLogger);
119        }
120        return logger;
121    }
122
123    @Override
124    public SchedulerAdapter getScheduler() {
125        if (scheduler == null) {
126            scheduler = new JavaScheduler();
127        }
128        return scheduler;
129    }
130
131    @Override
132    public PlatformUtil<Member, User, User> getPlatformUtil() {
133        if (platformUtil == null) {
134            platformUtil = new DiscordPlatformUtil(this);
135        }
136        return platformUtil;
137    }
138
139    @Override
140    public Thread getMainThread() {
141        if (mainThread == null) {
142            mainThread = Thread.currentThread();
143        }
144        return mainThread;
145    }
146
147    public boolean isEnabled() {
148        return isEnabled;
149    }
150
151    public boolean isShutdown() {
152        return shutdown;
153    }
154
155    private JDA setupJDA() {
156        try {
157            CommandClientBuilder client = new CommandClientBuilder();
158            DiscordOptions options = getDiscordOptions();
159            client.setOwnerId(options.getOwnerId());
160            client.setPrefix(options.getPrefix());
161            client.setServerInvite(options.getServerInvite());
162            client.setPrefixFunction(options.getPrefixFunction());
163            client.setCommandPreProcessBiFunction(options.getCommandPreProcessBiFunction());
164            this.jda = JDABuilder.createDefault(getToken()).addEventListeners(getListeners())
165                    .addEventListeners(client.build(), eventWaiter).enableIntents(getIntents())
166                    .setChunkingFilter(ChunkingFilter.ALL)
167                    .setMemberCachePolicy(MemberCachePolicy.ALL)
168                    .enableCache(
169                            Arrays.asList(CacheFlag.ACTIVITY, CacheFlag.EMOTE, CacheFlag.ROLE_TAGS))
170                    .setActivity(Activity.playing("Booting...")).setStatus(OnlineStatus.IDLE)
171                    .setHttpClient(Instances.HTTP_CLIENT).build().awaitReady();
172        } catch (Throwable thr) {
173            getPluginLogger().error("Error whilst trying to initialize JDA", thr);
174            onDisable();
175        }
176
177        return this.jda;
178    }
179
180    private void setupLogger() {
181        java.util.logging.Logger global = java.util.logging.Logger.getLogger("");
182        global.setUseParentHandlers(false);
183        for (java.util.logging.Handler handler : global.getHandlers()) {
184            global.removeHandler(handler);
185        }
186        org.apache.logging.log4j.core.Logger rootLogger =
187                (org.apache.logging.log4j.core.Logger) org.apache.logging.log4j.LogManager
188                        .getRootLogger();
189        global.addHandler(new ForwardLogHandler());
190        rootLogger.addFilter(new JDAFilter(this));
191        System.setOut(IoBuilder.forLogger(log4jLogger).setLevel(Level.INFO).buildPrintStream());
192        System.setErr(IoBuilder.forLogger(log4jLogger).setLevel(Level.WARN).buildPrintStream());
193    }
194
195    private void setupConsole() {
196        Thread thread = new Thread("Discord Console Handler") {
197            public void run() {
198                Console console = new net.deltapvp.library.jda.console.Console(DiscordBot.this);
199                console.start();
200            }
201        };
202        thread.setDaemon(true);
203        thread.start();
204    }

221
222    public JDA getJDA() {
223        if (this.jda == null) {
224            throw new RuntimeException("The JDA instance has not yet been initialized");
225        }
226        return this.jda;
227    }
