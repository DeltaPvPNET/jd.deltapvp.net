001package net.deltapvp.library.jda.console;
002
003import org.jline.reader.LineReader;
004import org.jline.reader.LineReaderBuilder;
005import net.deltapvp.library.common.logging.PluginLogger;
006import net.deltapvp.library.jda.DiscordBot;
007import net.minecrell.terminalconsole.SimpleTerminalConsole;
008
009public class Console extends SimpleTerminalConsole {
010
011    private final DiscordBot cli;
012    private final PluginLogger logger;
013
014    public Console(DiscordBot cli) {
015        this.cli = cli;
016        this.logger = cli.getPluginLogger();
017    }
018
019    @Override
020    protected boolean isRunning() {
021        return cli.isEnabled() && !cli.isShutdown();
022    }
023
024    @Override
025    protected void runCommand(String command) {
026        command = command.replaceFirst("/", "");
027        String[] args = command.split(" ");
028        switch (command.toLowerCase()) {
029            case "stop":
030            case "shutdown":
031            case "restart":
032            case "reboot":
033                shutdown();
034                break;
035            default:
036                logger.info(color("&cThe command /" + command + " could not be found."));
037        }
038    }
039
040    @Override
041    protected void shutdown() {
042        cli.onDisable();
043    }
044
045    @Override
046    protected LineReader buildReader(LineReaderBuilder builder) {
047        builder.appName("Discord")
048                .variable(LineReader.HISTORY_FILE, java.nio.file.Paths.get(".console_history"))
049                .option(LineReader.Option.COMPLETE_IN_WORD, true);
050        return super.buildReader(builder);
051    }
052
053    public static String color(String string) {
054        return string.replace("&", "ยง");
055    }
056}
