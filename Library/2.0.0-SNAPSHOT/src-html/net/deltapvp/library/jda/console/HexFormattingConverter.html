001package net.deltapvp.library.jda.console;
002
003import static net.minecrell.terminalconsole.MinecraftFormattingConverter.KEEP_FORMATTING_PROPERTY;
004import java.util.List;
005import java.util.regex.Matcher;
006import java.util.regex.Pattern;
007import org.apache.logging.log4j.core.LogEvent;
008import org.apache.logging.log4j.core.config.Configuration;
009import org.apache.logging.log4j.core.config.plugins.Plugin;
010import org.apache.logging.log4j.core.layout.PatternLayout;
011import org.apache.logging.log4j.core.pattern.ConverterKeys;
012import org.apache.logging.log4j.core.pattern.LogEventPatternConverter;
013import org.apache.logging.log4j.core.pattern.PatternConverter;
014import org.apache.logging.log4j.core.pattern.PatternFormatter;
015import org.apache.logging.log4j.core.pattern.PatternParser;
016import org.apache.logging.log4j.util.PerformanceSensitive;
017import org.apache.logging.log4j.util.PropertiesUtil;
018import net.minecrell.terminalconsole.TerminalConsoleAppender;
019
020@Plugin(name = "paperMinecraftFormatting", category = PatternConverter.CATEGORY)
021@ConverterKeys({"paperMinecraftFormatting"})
022@PerformanceSensitive("allocation")
023public final class HexFormattingConverter extends LogEventPatternConverter {
024
025    private static final boolean KEEP_FORMATTING =
026            PropertiesUtil.getProperties().getBooleanProperty(KEEP_FORMATTING_PROPERTY);
027
028    private static final String ANSI_RESET = "\u001B[m";
029
030    private static final char COLOR_CHAR = '§';
031    private static final String LOOKUP = "0123456789abcdefklmnor";
032
033    private static final String RGB_ANSI = "\u001B[38;2;%d;%d;%dm";
034    private static final Pattern NAMED_PATTERN =
035            Pattern.compile(COLOR_CHAR + "[0-9a-fk-orA-FK-OR]");
036    private static final Pattern RGB_PATTERN =
037            Pattern.compile(COLOR_CHAR + "x(" + COLOR_CHAR + "[0-9a-fA-F]){6}");
038
039    private static final String[] ansiCodes = new String[] {"\u001B[0;30m", // Black §0
040            "\u001B[0;34m", // Dark Blue §1
041            "\u001B[0;32m", // Dark Green §2
042            "\u001B[0;36m", // Dark Aqua §3
043            "\u001B[0;31m", // Dark Red §4
044            "\u001B[0;35m", // Dark Purple §5
045            "\u001B[0;33m", // Gold §6
046            "\u001B[0;37m", // Gray §7
047            "\u001B[0;30;1m", // Dark Gray §8
048            "\u001B[0;34;1m", // Blue §9
049            "\u001B[0;32;1m", // Green §a
050            "\u001B[0;36;1m", // Aqua §b
051            "\u001B[0;31;1m", // Red §c
052            "\u001B[0;35;1m", // Light Purple §d
053            "\u001B[0;33;1m", // Yellow §e
054            "\u001B[0;37;1m", // White §f
055            "\u001B[5m", // Obfuscated §k
056            "\u001B[21m", // Bold §l
057            "\u001B[9m", // Strikethrough §m
058            "\u001B[4m", // Underline §n
059            "\u001B[3m", // Italic §o
060            ANSI_RESET, // Reset §r
061    };
062
063    private final boolean ansi;
064    private final List<PatternFormatter> formatters;
065
066    /**
067     * Construct the converter.
068     *
069     * @param formatters The pattern formatters to generate the text to manipulate
070     * @param strip If true, the converter will strip all formatting codes
071     */
072    protected HexFormattingConverter(List<PatternFormatter> formatters, boolean strip) {
073        super("paperMinecraftFormatting", null);
074        this.formatters = formatters;
075        this.ansi = !strip;
076    }
077
078    @Override
079    public void format(LogEvent event, StringBuilder toAppendTo) {
080        int start = toAppendTo.length();
081        // noinspection ForLoopReplaceableByForEach
082        for (int i = 0, size = formatters.size(); i < size; i++) {
083            formatters.get(i).format(event, toAppendTo);
084        }
085
086        if (KEEP_FORMATTING || toAppendTo.length() == start) {
087            // Skip replacement if disabled or if the content is empty
088            return;
089        }
090
091        boolean useAnsi = ansi && TerminalConsoleAppender.isAnsiSupported();
092        String content = toAppendTo.substring(start);
093        content = useAnsi ? convertRGBColors(content) : stripRGBColors(content);
094        format(content, toAppendTo, start, useAnsi);
095    }
096
097    private static String convertRGBColors(String input) {
098        Matcher matcher = RGB_PATTERN.matcher(input);
099        StringBuffer buffer = new StringBuffer();
100        while (matcher.find()) {
101            String s = matcher.group().replace(String.valueOf(COLOR_CHAR), "").replace('x', '#');
102            int hex = Integer.decode(s);
103            int red = (hex >> 16) & 0xFF;
104            int green = (hex >> 8) & 0xFF;
105            int blue = hex & 0xFF;
106            String replacement = String.format(RGB_ANSI, red, green, blue);
107            matcher.appendReplacement(buffer, replacement);
108        }
109        matcher.appendTail(buffer);
110        return buffer.toString();
111    }
112
113    private static String stripRGBColors(String input) {
114        Matcher matcher = RGB_PATTERN.matcher(input);
115        StringBuffer buffer = new StringBuffer();
116        while (matcher.find()) {
117            matcher.appendReplacement(buffer, "");
118        }
119        matcher.appendTail(buffer);
120        return buffer.toString();
121    }
122
123    static void format(String content, StringBuilder result, int start, boolean ansi) {
124        int next = content.indexOf(COLOR_CHAR);
125        int last = content.length() - 1;
126        if (next == -1 || next == last) {
127            result.setLength(start);
128            result.append(content);
129            if (ansi) {
130                result.append(ANSI_RESET);
131            }
132            return;
133        }
134
135        Matcher matcher = NAMED_PATTERN.matcher(content);
136        StringBuffer buffer = new StringBuffer();
137        while (matcher.find()) {
138            int format = LOOKUP.indexOf(Character.toLowerCase(matcher.group().charAt(1)));
139            if (format != -1) {
140                matcher.appendReplacement(buffer, ansi ? ansiCodes[format] : "");
141            }
142        }
143        matcher.appendTail(buffer);
144
145        result.setLength(start);
146        result.append(buffer.toString());
147        if (ansi) {
148            result.append(ANSI_RESET);
149        }
150    }
151
152    /**
153     * Gets a new instance of the {@link HexFormattingConverter} with the specified options.
154     *
155     * @param config The current configuration
156     * @param options The pattern options
157     * @return The new instance
158     *
159     * @see HexFormattingConverter
160     */
161    public static HexFormattingConverter newInstance(Configuration config, String[] options) {
162        if (options.length < 1 || options.length > 2) {
163            LOGGER.error(
164                    "Incorrect number of options on paperMinecraftFormatting. Expected at least 1, max 2 received "
165                            + options.length);
166            return null;
167        }
168        if (options[0] == null) {
169            LOGGER.error("No pattern supplied on paperMinecraftFormatting");
170            return null;
171        }
172
173        PatternParser parser = PatternLayout.createPatternParser(config);
174        List<PatternFormatter> formatters = parser.parse(options[0]);
175        boolean strip = options.length > 1 && "strip".equals(options[1]);
176        return new HexFormattingConverter(formatters, strip);
177    }
