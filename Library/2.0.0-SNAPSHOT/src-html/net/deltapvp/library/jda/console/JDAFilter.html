017    private DiscordBot discord;
018    private String prefix = Console.color("&8[&6&lJDA&8] &e");
019    private List<String> set = Lists.newArrayList("net.deltapvp.library.lib.net.dv8tion.jda.api.",
020            "net.deltapvp.library.lib.net.dv8tion.jda.internal.",
021            "net.deltapvp.library.lib.com.jagrosh.jdautilities.");

023    public JDAFilter(DiscordBot discord) {
024        this.discord = discord;
025    }
026
027    public Result check(String loggerName, Level level, String message, Throwable throwable) {
028        // only listen for JDA logs
029        if (!check(loggerName))
030            return Result.NEUTRAL;
031        // if (!loggerName.contains("net.dv8tion.jda")) return Result.NEUTRAL;
032        switch (level.name()) {
033            case "INFO":
034                discord.getPluginLogger().info(prefix + message);
035                break;
036            case "WARN":
037                if (message.contains("Encountered 429")) {
038                    discord.getPluginLogger().debug(message);
039                    break;
040                }
041                discord.getPluginLogger().warn(prefix + message);
042                break;
043            case "ERROR":
044                if (message.contains("Requester timed out while executing a request")) {
045                    discord.getPluginLogger().error(prefix + message
046                            + ". This is either a issue on Discord's end (https://discordstatus.com) or with your server's connection");
047                    discord.getPluginLogger().debug(ExceptionUtils.getStackTrace(throwable));
048                    break;
049                }
050
051                if (throwable != null) {
052                    discord.getPluginLogger().error(
053                            prefix + message + "\n" + ExceptionUtils.getStackTrace(throwable));
054                } else {
055                    discord.getPluginLogger().error(prefix + message);
056                }
057                break;
058            case "DEBUG":
059                discord.getPluginLogger().debug(prefix + message);
060            default: // if (Debug.JDA.isVisible()) DiscordSRV.debug("[JDA] " + message);
061        }
062
063        // all JDA messages should be denied because we handle them ourselves
064        return Result.DENY;
065    }
066
067    public boolean check(String thing) {
068        thing = thing.toLowerCase();
069        for (String str : set) {
070            if (str.startsWith(thing)) {
071                return true;
072            }
073        }
074        for (String entry : set) {
075            entry = entry.toLowerCase();
076            if (entry.startsWith(thing) || entry.contains(thing) || entry.equalsIgnoreCase(thing)) {
077                return true;
078            }
079        }
080        return false;
081    }
082
083    @Override
084    public Result filter(LogEvent logEvent) {
085        return check(logEvent.getLoggerName(), logEvent.getLevel(),
086                logEvent.getMessage().getFormattedMessage(), logEvent.getThrown());
087    }
