001package net.deltapvp.library.jda.util;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.UUID;
006
007import net.deltapvp.library.common.util.PlatformUtil;
008import net.deltapvp.library.common.util.UUIDUtils;
009import net.deltapvp.library.jda.DiscordBot;
010import net.deltapvp.library.jda.util.DiscordUtil;
011import net.deltapvp.library.jda.util.SearchUtil;
012import net.dv8tion.jda.api.OnlineStatus;
013import net.dv8tion.jda.api.entities.Member;
014import net.dv8tion.jda.api.entities.User;
015import net.kyori.adventure.audience.Audience;
016
017public class DiscordPlatformUtil implements PlatformUtil<Member, User, User> {
018
019    protected final DiscordBot bot;
020
021    public DiscordPlatformUtil(DiscordBot bot) {
022        this.bot = bot;
023    }
024
025    @Override
026    public Member getPlayer(UUID uuid) {
027        return SearchUtil.getMember(bot.getMainGuild(), UUIDUtils.getName(uuid));
028    }
029
030    @Override
031    public Member getPlayer(String name) {
032        return SearchUtil.getMember(bot.getMainGuild(), name);
033    }
034
035    @Override
036    public User getCommandSender(UUID uuid) {
037        return SearchUtil.getMember(bot.getMainGuild(), UUIDUtils.getName(uuid)).getUser();
038    }
039
040    @Override
041    public User getConsole() {
042        return bot.getJDA().getSelfUser();
043    }
044
045    @Override
046    public Audience getAudience(UUID uuid) {
047        return Audience.empty();
048    }
049
050    @Override
051    public List<Audience> getAudiences() {
052        return Collections.singletonList(Audience.empty());
053    }
054
055    @Override
056    public boolean isPluginEnabled(String name) {
057        return false;
058    }
059
060    @Override
061    public boolean isOnline(UUID uuid) {
062        return false;
063    }
064
065    @Override
066    public boolean isOnline(String name) {
067        return SearchUtil.getMember(bot.getMainGuild(), name).getOnlineStatus() != OnlineStatus.OFFLINE;
068    }
069
070    @Override
071    public UUID getPlayerUUID(String name) {
072        return UUIDUtils.uuid(name);
073        //return DiscordUtil.getDiscriminatedName(SearchUtil.getMember(Discord.getInstance().getMainGuild(), name));
074    }
075
076    @Override
077    public String getPlayerName(UUID uuid) {
078        return DiscordUtil.getDiscriminatedName(SearchUtil.getMember(bot.getMainGuild(), UUIDUtils.getName(uuid)));
079    }
080
081    @Override
082    public void rebootLogic(int seconds) {
083        throw new UnsupportedOperationException("Cannot do reboot logic on Discord");
084        
085    }
086
087    @Override
088    public boolean isWhitelisted() {
089        return false;
090    }
091
092    @Override
093    public void setWhitelisted(boolean value) {
094        throw new UnsupportedOperationException("Cannot enable whitelist on Discord");
095        
096    }
097
098    @Override
099    public void performCommand(UUID uuid, String command) {
100        throw new UnsupportedOperationException("Cannot execute a command on Discord");
101        
102    }
103    
104}
