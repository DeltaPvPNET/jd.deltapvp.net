025package net.deltapvp.library.velocity.platform;
026
027import java.util.Collections;
028import java.util.List;
029import java.util.Optional;
030import java.util.UUID;
031
032import com.velocitypowered.api.command.CommandSource;
033import com.velocitypowered.api.proxy.ConsoleCommandSource;
034import com.velocitypowered.api.proxy.Player;
035
036import net.deltapvp.library.common.util.PlatformUtil;
037import net.deltapvp.library.common.util.UUIDUtils;
038import net.deltapvp.library.velocity.Library;
039import net.kyori.adventure.audience.Audience;
040
041public class VelocityPlatformUtil implements PlatformUtil<Player, CommandSource, ConsoleCommandSource> {
042
043    @Override
044    public Player getPlayer(UUID uuid) {
045        return Library.getInstance().getProxy().getPlayer(uuid).orElse(null);
046    }
047
048    @Override
049    public Player getPlayer(String name) {
050        return Library.getInstance().getProxy().getPlayer(name).orElse(null);
051    }
052
053    @Override
054    public CommandSource getCommandSender(UUID uuid) {
055        if (getPlayer(uuid) == null) return null;
056        if (getPlayer(uuid) instanceof CommandSource) {
057            return ((CommandSource) getPlayer(uuid));
058        } else {
059            return null;
060        }
061    }
062
063    @Override
064    public ConsoleCommandSource getConsole() {
065        return Library.getInstance().getProxy().getConsoleCommandSource();
066    }
067
068    @Override
069    public Audience getAudience(UUID uuid) {
070        return getCommandSender(uuid);
071    }
072
073    @Override
074    public List<Audience> getAudiences() {
075        return Collections.emptyList();
076    }
077
078    @Override
079    public boolean isPluginEnabled(String name) {
080        return Library.getInstance().getProxy().getPluginManager().isLoaded(name.toLowerCase());
081    }
082
083    @Override
084    public boolean isOnline(UUID uuid) {
085        return Library.getInstance().getProxy().getPlayer(uuid).isPresent();
086    }
087
088    @Override
089    public boolean isOnline(String name) {
090        return Library.getInstance().getProxy().getPlayer(name).isPresent();
091    }
092
093    @Override
094    public UUID getPlayerUUID(String name) {
095        Optional<Player> plr = Library.getInstance().getProxy().getPlayer(name);
096        return plr.isPresent() ? plr.get().getUniqueId() : null;
097    }
098
099    @Override
100    public String getPlayerName(UUID uuid) {
101        Optional<Player> plr = Library.getInstance().getProxy().getPlayer(uuid);
102        return plr.isPresent() ? plr.get().getUsername() : null;
103    }
104    @Override
105    public void rebootLogic(int seconds) {
106        // TODO Auto-generated method stub
107        
108    }
109
110    @Override
111    public boolean isWhitelisted() {
112        // TODO Auto-generated method stub
113        return false;
114    }
115
116    @Override
117    public void setWhitelisted(boolean value) {
118        // TODO Auto-generated method stub
119        
120    }
121
122    @Override
123    public void performCommand(UUID uuid, String command) {
124        if (UUIDUtils.isConsoleOrNull(uuid)) {
125            Library.getInstance().getProxy().getCommandManager().executeAsync(getConsole(), command);
126        } else {
127            Library.getInstance().getProxy().getCommandManager().executeAsync(getCommandSender(uuid), command);
128        }
129        
130    }
131    
132}
