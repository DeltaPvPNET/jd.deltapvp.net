025package net.deltapvp.library.common.config.util;
026
027import org.spongepowered.configurate.CommentedConfigurationNode;
028import org.spongepowered.configurate.serialize.SerializationException;
029import net.deltapvp.library.common.config.impl.ConfigurateConfigFile;
030import java.math.BigDecimal;
031import java.math.MathContext;
032import java.util.ArrayList;
033import java.util.Collections;
034import java.util.LinkedHashMap;
035import java.util.LinkedHashSet;
036import java.util.List;
037import java.util.Map;
038import java.util.Set;
039
040public final class ConfigurateUtil {
041    private ConfigurateUtil() {
042    }
043
044    public static Set<String> getRootNodeKeys(final ConfigurateConfigFile config) {
045        return getKeys(config.getRootNode());
046    }
047
048    public static Set<String> getKeys(final CommentedConfigurationNode node) {
049        if (node == null || !node.isMap()) {
050            return Collections.emptySet();
051        }
052
053        final Set<String> keys = new LinkedHashSet<>();
054        for (Object obj : node.childrenMap().keySet()) {
055            keys.add(String.valueOf(obj));
056        }
057        return keys;
058    }
059
060    public static Map<String, CommentedConfigurationNode> getMap(final CommentedConfigurationNode node) {
061        if (node == null || !node.isMap()) {
062            return Collections.emptyMap();
063        }
064
065        final Map<String, CommentedConfigurationNode> map = new LinkedHashMap<>();
066        for (Map.Entry<Object, CommentedConfigurationNode> entry : node.childrenMap().entrySet()) {
067            map.put(String.valueOf(entry.getKey()), entry.getValue());
068        }
069        return map;
070    }
071
072    public static Map<String, Object> getRawMap(final ConfigurateConfigFile config, final String key) {
073        if (config == null || key == null) {
074            return Collections.emptyMap();
075        }
076        return getRawMap(config.getSection(key));
077    }
078
079    public static Map<String, Object> getRawMap(final CommentedConfigurationNode node) {
080        if (node == null || !node.isMap()) {
081            return Collections.emptyMap();
082        }
083
084        final Map<String, Object> map = new LinkedHashMap<>();
085        for (Map.Entry<Object, CommentedConfigurationNode> entry : node.childrenMap().entrySet()) {
086            map.put(String.valueOf(entry.getKey()), entry.getValue().raw());
087        }
088        return map;
089    }
090
091    public static List<Map<?, ?>> getMapList(final CommentedConfigurationNode node) {
092        List<?> list = null;
093        try {
094            list = node.getList(Object.class);
095        } catch (SerializationException ignored) {
096        }
097        final List<Map<?, ?>> result = new ArrayList<>();
098
099        if (list == null) {
100            return result;
101        }
102
103        for (Object object : list) {
104            if (object instanceof Map) {
105                result.add((Map<?, ?>) object);
106            }
107        }
108        return result;
109    }
110
111    public static BigDecimal toBigDecimal(final String input, final BigDecimal def) {
112        if (input == null || input.isEmpty()) {
113            return def;
114        }
115
116        try {
117            return new BigDecimal(input, MathContext.DECIMAL128);
118        } catch (final NumberFormatException | ArithmeticException e) {
119            return def;
120        }
121    }
122
123    public static boolean isDouble(final CommentedConfigurationNode node) {
124        return node != null && node.raw() instanceof Double;
125    }
126
127    public static boolean isInt(final CommentedConfigurationNode node) {
128        return node != null && node.raw() instanceof Integer;
129    }
130
131    public static boolean isString(final CommentedConfigurationNode node) {
132        return node != null && node.raw() instanceof String;
133    }
134}
