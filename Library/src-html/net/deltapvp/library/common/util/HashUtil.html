025package net.deltapvp.library.common.util;
026
027import java.time.Duration;
028
029import com.github.benmanes.caffeine.cache.Cache;
030import com.password4j.PBKDF2Function;
031import com.password4j.Password;
032import com.password4j.types.Hmac;
033
034import org.jetbrains.annotations.ApiStatus;
035import org.jetbrains.annotations.NotNull;
036import org.jetbrains.annotations.Nullable;
037
038@ApiStatus.AvailableSince("2.0.0")
039/**
040 * @since 2.0.0
041 * @author powercas_gamer
042 * Utility class to hash strings
043 */
044public class HashUtil {
045    private final static String PEPER = "12345";
046    private final static PBKDF2Function FUNCTION = PBKDF2Function.getInstance(Hmac.SHA512, 64000, 512);
047    public static final Cache<String, String> CACHE = CaffeineFactory.newBuilder()
048        .softValues()
049        .expireAfterWrite(Duration.ofMinutes(30))
050        .expireAfterAccess(Duration.ofMinutes(30))
051        .build();
052
053    /**
054     * Hash a string with the default peper
055     * 
056     * @param string the string to hash
057     * @return the hashed string
058     */
059    public static String hashString(@NotNull String string) {
060        return hashString(string, null);
061    }
062
063    /**
064     * Hash a string with the possibility of a custom peper
065     * 
066     * @param string the string to hash
067     * @param peper the peper to use or else the default peper
068     * @return the hashed string
069     */
070    public static String hashString(@NotNull String string, @Nullable String peper) {
071        return CACHE.get(string, $$ -> {
072            return Password.hash(string)
073                .addPepper(peper == null ? PEPER : peper)
074                .addRandomSalt()
075                .with(FUNCTION)
076                .getResult();
077        });
078    }
079}
