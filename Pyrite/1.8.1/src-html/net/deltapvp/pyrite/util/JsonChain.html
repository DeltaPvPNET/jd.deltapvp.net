025package net.deltapvp.pyrite.util;
026
027import java.util.List;
028import java.util.Map;
029import java.util.Optional;
030import java.util.Set;
031import java.util.UUID;
032
033import com.google.gson.JsonElement;
034import com.google.gson.JsonObject;
035
036import net.deltapvp.pyrite.Pyrite;
037
038public class JsonChain {
039
040    private final JsonObject json = new JsonObject();
041
042    public JsonChain addProperty(String property, String value) {
043        this.json.addProperty(property, value);
044        return this;
045    }
046
047    public JsonChain addProperty(String property, Number value) {
048        this.json.addProperty(property, value);
049        return this;
050    }
051
052    public JsonChain addProperty(String property, Boolean value) {
053        this.json.addProperty(property, value);
054        return this;
055    }
056
057    public JsonChain addProperty(String property, Character value) {
058        this.json.addProperty(property, value);
059        return this;
060    }
061
062    public JsonChain addProperty(String property, Object value) {
063        this.json.addProperty(property, Pyrite.GSON.toJson(value));
064        return this;
065    }
066
067    public JsonChain addProperty(String property, UUID value) {
068        this.json.addProperty(property, FastUUID.toString(value));
069        return this;
070    }
071
072    public <U> JsonChain addProperty(String property, List<U> value) {
073        this.json.addProperty(property, Pyrite.GSON.toJson(value));
074        return this;
075    }
076
077    public <U> JsonChain addProperty(String property, Set<U> value) {
078        this.json.addProperty(property, Pyrite.GSON.toJson(value));
079        return this;
080    }
081
082    public <K, V> JsonChain addProperty(String property, Map<K, V> value) {
083        this.json.addProperty(property, Pyrite.GSON.toJson(value));
084        return this;
085    }
086
087    public <U> JsonChain addProperty(String property, Optional<U> value) {
088        return addProperty(property, value, true);
089    }
090
091    public <U> JsonChain addProperty(String property, Optional<U> value, boolean serializeNull) {
092        if (serializeNull) {
093            this.json.addProperty(property, value.isPresent() ? Pyrite.GSON.toJson(value.get()) : null);
094        } else {
095            value.ifPresent(v -> this.json.addProperty(property, Pyrite.GSON.toJson(v)));
096        }
097        return this;
098    }
099
100    public JsonChain add(String property, JsonElement element) {
101        this.json.add(property, element);
102        return this;
103    }
104
105    public JsonChain add(String property, JsonObject object) {
106        this.json.add(property, object);
107        return this;
108    }
109
110    public JsonObject get() {
111        return this.json;
112    }
113
114    public String toJson() {
115        return Pyrite.GSON.toJson(get());
116    }
117
118}
