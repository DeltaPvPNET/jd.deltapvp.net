025package net.deltapvp.library.common.uuid.types;
026
027import java.util.UUID;
028import java.util.Map.Entry;
029
030import com.github.benmanes.caffeine.cache.Cache;
031import com.mongodb.client.model.Filters;
032import com.mongodb.client.model.ReplaceOptions;
033
034import org.bson.Document;
035import org.jetbrains.annotations.NotNull;
036import org.jetbrains.annotations.Nullable;
037
038import net.deltapvp.library.common.LibraryShared;
039import net.deltapvp.library.common.util.CaffeineFactory;
040import net.deltapvp.library.common.uuid.UUIDMapping;
041import net.deltapvp.library.common.uuid.UUIDService;
042
043public class MongoUUIDService implements UUIDService {
044    public Cache<UUID, String> uuidToName = CaffeineFactory.newBuilder().build();
045    public Cache<String, UUID> nameToUuid = CaffeineFactory.newBuilder().build();
046
047    public MongoUUIDService() {
048    }
049
050    @Override
051    public @NotNull String getServiceName() {
052        return "MongoDB Cache";
053    }
054
055    @Override
056    public @Nullable String getName(UUID uuid) {
057        return uuidToName.get(uuid, k -> {
058            String name = null;
059            Document doc = LibraryShared.getMongoHandler().getCacheCollection().find(Filters.eq("uuid", uuid)).first();
060            if (doc == null) return null;
061            UUIDMapping map = UUIDMapping.fromDocument(doc);
062
063            name = map.getName();
064
065            return name;
066        });
067    }
068
069    @Override
070    public @Nullable UUID getUUID(String name) {
071        return nameToUuid.get(name, k -> {
072            UUID uuid = null;
073            Document doc = LibraryShared.getMongoHandler().getCacheCollection().find(Filters.eq("name", name)).first();
074            if (doc == null) return null;
075            UUIDMapping map = UUIDMapping.fromDocument(doc);
076
077            uuid = map.getUUID();
078
079            return uuid;
080        });
081    }
082                                                                                                                                                                                        
083    @Override
084    public void update(UUID uuid, String name) {
085        UUIDMapping mapping = new UUIDMapping(uuid, name);
086        nameToUuid.put(name, uuid);
087        uuidToName.put(uuid, name);
088        LibraryShared.getMongoHandler().getCacheCollection().replaceOne(Filters.eq("uuid", uuid), mapping.toDocument(), new ReplaceOptions().upsert(true));
089    }    
090
091    @Override
092    public void updateAll(UUID uuid, String name) {
093        update(uuid, name);
094
095        // send packet
096
097     //   LibraryShared.getMongoHandler().getCacheCollection().replaceOne(Filters.eq("uuid", uuid), new UUIDMapping(uuid, name).toDocument(), new ReplaceOptions().upsert(true));
098    }
099
100    @Override
101    public void onShutdown() {
102        for (Entry<String, UUID> entry : this.nameToUuid.asMap().entrySet()) {
103         //   updateAll(entry.getValue(), entry.getKey());
104        }
105    }
106}
